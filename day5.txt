
- name: copy index to destination
  template: src=index.html dest=/home/ubuntu
  register: copy_result
  ignore_errors: true

- name: check the condition
  debug:
    var: copy_result
  
- name: index.html copy with failed message
  template: src=index.html dest=/home/ubuntu
  failed_when:
    - copy_result.failed == true
    - '"Could not find or access in "copy_result.msg'

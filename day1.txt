
#declare an array

use_numbers=[]
print(f"Initial Value: {use_numbers}")

data= int(input('How many values you need to add? '))
print(f"You Entered the range Value: {data}")

# take the value

for i in range(data):
    num=int(input (f"Enter {i+1} value: "))
    use_numbers.append(num)
print(use_numbers)

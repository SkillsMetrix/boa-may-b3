- hosts: web
  become: true
  tasks:
    - name: install apache server on web
      apt:
        name:
          - apache2
        update_cache: yes
        state: latest

    - name: start apache service
      service:
        name: apache2
        state: started
      
    - name: check installation service
      command: systemctl status apache2
      register: apache_status

    - name: display the value
      debug:
        msg: "{{apache_status.stdout}}"
